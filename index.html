⚙️ Configurar Google Apps Script

O erro de conexão ocorre porque o Apps Script precisa suportar JSONP e estar implantado com acesso público. Siga os passos:

  1. Abra o script.google.com e abra seu projeto
  2. Substitua o conteúdo do Código.gs pelo código abaixo (se já for igual, pule)
  3. Clique em Implantar → Gerenciar implantações → Editar (lápis)
  4. Em "Versão" escolha Nova versão
  5. Executar como: Eu | Quem tem acesso: Qualquer pessoa
  6. Clique em Implantar e copie a nova URL abaixo
// ── ConectaPlac – Código.gs (versão correta) ─────────────────
// Cole este código no Apps Script e reimplante como App da Web
// Executar como: Eu  |  Quem tem acesso: Qualquer pessoa

const API_KEY = "conectaplac@2026uniceplac";

function json(obj) {
  return ContentService
    .createTextOutput(JSON.stringify(obj))
    .setMimeType(ContentService.MimeType.JSON);
}

function chaveValida(chave) {
  return typeof chave === "string" && chave === API_KEY;
}

function doGet(e) {
  const chave = e && e.parameter && e.parameter.apiKey;
  if (!chaveValida(chave))
    return json({ status: "error", code: 401, message: "Acesso não autorizado." });

  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("Página1");
  const valores = sheet.getDataRange().getValues();
  const resultado = [];
  for (let i = 1; i < valores.length; i++) {
    const l = valores[i];
    if (!l[1] && !l[2]) continue;
    resultado.push({ id:l[0], nome:l[1], cpf:l[2], nascimento:l[3],
      telefone:l[4], endereco:l[5], servico:l[6], obs:l[7], data:l[8],
      dataCadastro:l[9], dataAgendamento:l[10], horarioAgendamento:formatarHorario(l[11]) });
  }
  return ContentService.createTextOutput(JSON.stringify(resultado))
    .setMimeType(ContentService.MimeType.JSON);
}

function doPost(e) {
  const dados = JSON.parse(e.postData.contents);
  if (!chaveValida(dados.apiKey))
    return json({ status: "error", code: 401, message: "Acesso não autorizado." });

  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("Página1");

  if (dados.action === "deleteAll") {
    const lr = sheet.getLastRow();
    if (lr > 1) sheet.deleteRows(2, lr - 1);
    return json({ status: "success" });
  }
  if (dados.action === "delete") {
    const v = sheet.getDataRange().getValues();
    for (let i = 1; i < v.length; i++) {
      if (String(v[i][0]) === String(dados.id)) { sheet.deleteRow(i+1); break; }
    }
    return json({ status: "success" });
  }
  sheet.appendRow([dados.id||Date.now(), dados.nome||"", dados.cpf||"",
    dados.nascimento||"", dados.telefone||"", dados.endereco||"",
    dados.servico||"Não informado", dados.obs||"—", dados.data||"",
    dados.dataCadastro||new Date().toLocaleString('pt-BR'),
    dados.dataAgendamento||"—", dados.horarioAgendamento||"—"]);
  return json({ status: "success" });
}

function formatarHorario(valor) {
  if (!valor || valor === "—" || valor === "") return "—";
  try {
    return Utilities.formatDate(new Date(valor), Session.getScriptTimeZone(), "HH:mm");
  } catch(e) { return String(valor); }
}
